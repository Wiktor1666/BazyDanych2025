Zadanie 1.
1)	SELECT c.FULL_NAME, c_a.STREET, LENGTH(c_a.STREET) AS DLUGOSC_ULICY, SUBSTR(c_a.STREET, 1, 5) AS PIERWSZE_5_LITER FROM CLIENT c JOIN CLIENT_ADDRESS c_a ON c.CLIENT_ID = c_a.CLIENT_ID;

2)	SELECT c.SHORT_NAME, c_a.ADDRESS_TYPE FROM CLIENT c LEFT JOIN CLIENT_ADDRESS c_a ON c.CLIENT_ID = c_a.CLIENT_ID;

3)	SELECT p.PRODUCT_NAME, c.CATEGORY_NAME, SUBSTR(p.PRODUCT_NAME, 1, 3) AS PREFIX FROM PRODUCT p JOIN CATEGORY c ON p.CATEGORY_ID = c.CATEGORY_ID;

4)	SELECT p.PRODUCT_NAME, p.DESCRIPTION FROM PRODUCT p JOIN ORDER_POSITION o_p ON p.PRODUCT_ID = o_p.PRODUCT_ID GROUP BY p.PRODUCT_ID, p.PRODUCT_NAME, p.DESCRIPTION;


Zadanie 2.
1)	SELECT c.FULL_NAME, `ORDER`.ORDER_NUMBER, `ORDER`.ORDER_DATE, DATEDIFF(CURDATE(), `ORDER`.ORDER_DATE) AS DNI_OD_ZAMOWIENIA FROM `ORDER` JOIN CLIENT c ON `ORDER`.CLIENT_ID = c.CLIENT_ID;

2)	SELECT `ORDER`.ORDER_ID, `ORDER`.ORDER_DATE, DATE_ADD(`ORDER`.ORDER_DATE, INTERVAL 14 DAY) AS TERMIN_PLATNOSCI FROM `ORDER` WHERE `ORDER`.ORDER_STATUS NOT IN (3, 7, 10);

3)	SELECT e.FIRST_NAME, e.LAST_NAME, e.DATE_OF_EMPLOYMENT, DAYNAME(e.DATE_OF_EMPLOYMENT) AS DZIEN_TYGODNIA, d.DEPARTMENT_NAME FROM EMPLOYEE e LEFT JOIN DEPARTMENT d ON e.DEPARTMENT_ID = d.DEPARTMENT_ID;

4)	SELECT c.FULL_NAME, (DATEDIFF(MAX(`ORDER`.ORDER_DATE), MIN(`ORDER`.ORDER_DATE))) AS ROZNICA_DNI FROM CLIENT c LEFT JOIN `ORDER` ON c.CLIENT_ID = `ORDER`.CLIENT_ID GROUP BY c.CLIENT_ID, c.FULL_NAME;


Zadanie 3.
1)	SELECT c.SHORT_NAME, GROUP_CONCAT(c_a.STREET SEPARATOR ', ') AS ADRESY FROM CLIENT c LEFT JOIN CLIENT_ADDRESS c_a ON c.CLIENT_ID = c_a.CLIENT_ID GROUP BY c.CLIENT_ID, c.SHORT_NAME;

2)	SELECT CITY, GROUP_CONCAT(DISTINCT SUBSTR(POSTAL_CODE, 1, 2) SEPARATOR ', ') AS PREFIXY FROM CLIENT_ADDRESS GROUP BY CITY;

3)	SELECT CONCAT(e.LAST_NAME, ' ', e.FIRST_NAME) AS PELNA_NAZWA, d.DEPARTMENT_NAME, DATEDIFF(CURDATE(), e.DATE_OF_EMPLOYMENT) AS DNI_OD_ZARUDNIENIA FROM EMPLOYEE e LEFT JOIN DEPARTMENT d ON e.DEPARTMENT_ID = d.DEPARTMENT_ID;

4)	SELECT CONCAT(c.FULL_NAME, ' - ', c.SHORT_NAME) AS NAZWA, GROUP_CONCAT(RIGHT(`ORDER`.ORDER_NUMBER, 3) SEPARATOR ', ') AS SUFFIX_ZAMOWIENIA FROM CLIENT c LEFT JOIN `ORDER` ON c.CLIENT_ID = `ORDER`.CLIENT_ID GROUP BY c.CLIENT_ID, c.FULL_NAME, c.SHORT_NAME;

5)	SELECT e.FIRST_NAME, e.LAST_NAME, CONCAT(LOWER(SUBSTR(e.FIRST_NAME, 1, 3)), UPPER(RIGHT(e.LAST_NAME, 4)), LENGTH(e.FIRST_NAME) + LENGTH(c_a.STREET)) AS HASLO FROM EMPLOYEE e JOIN CLIENT_ADDRESS c_a ON e.EMPLOYEE_NAME = c_a.CLIENT_ID;
