Zadanie 1.
1)	SELECT * FROM CLIENT WHERE CLIENT_ID NOT IN (SELECT CLIENT_ID FROM `ORDER`);
	SELECT c.* FROM `CLIENT` c LEFT JOIN `ORDER` o ON c.CLIENT_ID = o.CLIENT_ID WHERE o.ORDER_ID IS NULL;

2)	SELECT u.UNIT_NAME FROM `INVENTORY` i RIGHT JOIN `UNIT_OF_MEASUREMENT` u ON i.UNIT_ID = u.UNIT_ID WHERE i.PRODUCT_ID IS NULL;

3)	SELECT p.PRODUCT_NAME, AVG(op.SELL_PRICE - p.PURCHASE_PRICE) AS SREDNIA_ROZNICA FROM `ORDER_POSITION` op JOIN `PRODUCT` p ON op.PRODUCT_ID = p.PRODUCT_ID GROUP BY p.PRODUCT_ID, p.PRODUCT_NAME ORDER BY SREDNIA_ROZNICA DESC;


Zadanie 2.
1)	SELECT D d.DEPARTMENT_NAME FROM `DEPARTMENT` d JOIN `EMPLOYEE` e ON d.DEPARTMENT_ID = e.DEPARTMENT_ID JOIN `ORDER` o ON e.EMPLOYEE_NAME = o.EMPLOYEE_ID;

2)	SELECT d.DEPARTMENT_NAME, SUM(op.AMOUNT * op.SELL_PRICE) AS LACZNA_WARTOSC FROM `DEPARTMENT` d JOIN `EMPLOYEE` e ON d.DEPARTMENT_ID = e.DEPARTMENT_ID JOIN `ORDER` o ON e.EMPLOYEE_NAME = o.EMPLOYEE_ID JOIN `ORDER_POSITION` op ON o.ORDER_ID = op.ORDER_ID GROUP BY d.DEPARTMENT_NAME;

3)	SELECT d.DEPARTMENT_NAME, os.STATUS_NAME, SUM(op.AMOUNT * op.SELL_PRICE) AS LACZNA_WARTOSC FROM `DEPARTMENT` d JOIN `EMPLOYEE` e ON d.DEPARTMENT_ID = e.DEPARTMENT_ID JOIN `ORDER` o ON e.EMPLOYEE_NAME = o.EMPLOYEE_ID JOIN `ORDER_POSITION` op ON o.ORDER_ID = op.ORDER_ID JOIN `ORDER_STATUS` os ON o.ORDER_STATUS = os.STATUS_ID GROUP BY d.DEPARTMENT_NAME, os.STATUS_NAME ORDER BY d.DEPARTMENT_NAME ASC, LACZNA_WARTOSC DESC;


Zadanie 3.
1)	SELECT CONCAT(PRODUCT_NAME, ' (', DESCRIPTION, ')') AS PELNA_NAZWA FROM `PRODUCT`;

2)	SELECT CONCAT_WS(',', PRODUCT_ID, CATEGORY_ID, PRODUCT_NAME, PURCHASE_PRICE, DESCRIPTION) AS PRODUKTY FROM `PRODUCT`;

3)	SELECT d.DEPARTMENT_NAME, GROUP_CONCAT(CONCAT(e.FIRST_NAME, ' ', e.LAST_NAME) SEPARATOR ', ') AS PRACOWNICY FROM `DEPARTMENT` d JOIN `EMPLOYEE` e ON d.DEPARTMENT_ID = e.DEPARTMENT_ID GROUP BY d.DEPARTMENT_NAME;

4)	SELECT c.CATEGORY_NAME, GROUP_CONCAT(p.PRODUCT_NAME SEPARATOR ', ') AS NIESPRZEDANE_PRODUKTY FROM `PRODUCT` p JOIN `CATEGORY` c ON p.CATEGORY_ID = c.CATEGORY_ID LEFT JOIN `ORDER_POSITION` op ON p.PRODUCT_ID = op.PRODUCT_ID p.POSITION_ID IS NULL GROUP BY c.CATEGORY_NAME;


Zadanie 4.
1)	SELECT ORDER_NUMBER, IF(ORDER_STATUS IN (3, 4, 7, 10), 'Zakonczone', 'Inny') AS STATUS FROM `ORDER`;

2)	SELECT p.PRODUCT_NAME, CASE WHEN i.AMOUNT = 0 THEN 'Brak' WHEN i.AMOUNT > 0 AND i.AMOUNT < 10 THEN 'Mala ilosc' WHEN i.AMOUNT > 9 AND i.AMOUNT <= 30 THEN 'Srednia ilosc' WHEN i.AMOUNT > 30 THEN 'Duza ilosc' END AS dostepnosc FROM PRODUCT p JOIN INVENTORY i ON p.PRODUCT_ID = i.PRODUCT_ID;

3)	SELECT FULL_NAME, IFNULL(TAX_IDENTIFIER, 'Klient indywidualny') AS IDENTYFIKATOR FROM CLIENT;


Zadanie 5.
1)	SELECT CITY, SUBSTR(POSTAL_CODE, 1, 2) AS KOD_REGIONU FROM CLIENT_ADDRESS;

2)	SELECT CITY, GROUP_CONCAT(DISTINCT SUBSTR(POSTAL_CODE, 1, 2) SEPARATOR ', ') AS KODY_W_MIESCIE FROM CLIENT_ADDRESS GROUP BY CITY;

3)	SELECT TRIM(SUBSTR(STREET, 5, LENGTH(STREET) - 4 - LENGTH(SUBSTRING_INDEX(STREET, ' ', -1)))) AS NAZWA_ULICY FROM CLIENT_ADDRESS;
